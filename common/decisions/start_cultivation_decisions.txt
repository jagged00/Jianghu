# active Roots

open_meridians_decision = {

	picture = {
		reference = "gfx/interface/illustrations/decisions/decision_find_roots.dds"
	}
	
	#decision_group_type = cultivation
	sort_order = 500

    desc = open_meridians_decision_desc
    selection_tooltip = open_meridians_decision_selection_tooltip
	ai_goal = yes

	cooldown = { years = 10 }

    is_shown = {
        NOT = {
			has_trait = active_meridians
		}
		prowess >= 10
    }
	is_valid = {
		OR = {
			short_term_gold >= 500
			AND = {
				OR = {
					has_trait = good_meridians
					has_trait = excellent_meridians
					has_trait = heavenly_meridians
				}
				short_term_gold >= 150
			}
		}
	}
    ai_check_interval = 60

    
	ai_will_do = {
		base = 100
		modifier = {
			add = chance_to_cultivate
			is_adult = yes
		}
	}

    effect = {
		if = {
			limit = {
				OR = {
					has_trait = good_meridians
					has_trait = excellent_meridians
					has_trait = heavenly_meridians
				}
			}
			remove_short_term_gold = 150
		}
		else = {
			remove_short_term_gold = 500
		}
		custom_tooltip = open_meridians_decision_tooltip
        trigger_event = cultivation_important_event.001
	}
}


#Begin Cultivation

begin_cultivation_decision = {

	picture = {
		reference = "gfx/interface/illustrations/decisions/decision_start_cultivation.dds"
	}

    desc = begin_cultivation_decision_desc
    selection_tooltip = begin_cultivation_decision_selection_tooltip
	sort_order = 501

    cooldown = { months = 1 }

    is_shown = {
		AND = {
			is_adult = yes
			has_trait = active_meridians
			has_character_flag = joined_any_cultivation_sect
			NOT = { has_character_flag = cultivation_initiated }
			NOT = {
				has_trait = cultivation_realm
			}

		}
    }
	is_valid = {
		OR = {
			short_term_gold >= 300
			AND = {
				OR = {
					has_trait = good_meridians
					has_trait = excellent_meridians
					has_trait = heavenly_meridians
				}
				short_term_gold >= 100
			}
		}
	}
    ai_check_interval = 60

	ai_will_do = {
		base = 100
		modifier = {
			add = chance_to_cultivate
			is_adult = yes
		}
	}


    effect = {
		if = {
			limit = {
				OR = {
					has_trait = good_meridians
					has_trait = excellent_meridians
					has_trait = heavenly_meridians
				}
			}
			remove_short_term_gold = 100
		}
		else = {
			remove_short_term_gold = 300
		}
		add_character_modifier = {
			modifier = unlocking_cultivation_modifier
			months = 1
		}

        trigger_event = {
			id = cultivation_important_event.002
			months =  1
		}
		custom_tooltip = begin_cultivation_decision_tooltip

		
    }
}
