breakthrough_effect = {
    if = {
        limit = { has_trait = qi_gathering }
        remove_trait = qi_gathering
        add_trait = foundation_establishment
    }
    else_if = {
        limit = { has_trait = foundation_establishment }
        remove_trait = foundation_establishment
        add_trait = core_formation
    }
    else_if = {
        limit = { has_trait = core_formation }
        remove_trait = core_formation
        add_trait = nascent_soul
    }
    else_if = {
        limit = { has_trait = nascent_soul }
        remove_trait = nascent_soul
        add_trait = spirit_severing
    }
    else_if = {
        limit = { has_trait = spirit_severing }
        remove_trait = spirit_severing
        add_trait = dao_seeking
    }
    else_if = {
        limit = { has_trait = dao_seeking }
        remove_trait = dao_seeking
        add_trait = immortal_ascension
    }
}

improve_meridians = {
    if = {
        limit = { has_trait = poor_meridians }
        remove_trait = poor_meridians
        add_trait = normal_meridians
    }
    else_if = {
        limit = { has_trait = normal_meridians }
        remove_trait = normal_meridians
        add_trait = good_meridians
    }
    else_if = {
        limit = { has_trait = good_meridians }
        remove_trait = good_meridians
        add_trait = excellent_meridians
    }
    else_if = {
        limit = { has_trait = excellent_meridians }
        remove_trait = excellent_meridians
        add_trait = heavenly_meridians
    }
    else = {
        earn_cultivation_xp = yes
    }
}

wound_cultivation_character = {
    if = {
        limit = {
            NOT = {
                has_trait = wounded
            }
        }
        add_trait = wounded_1
    }
    else_if = {
        limit = { has_trait = wounded_1 }
        remove_trait = wounded_1
        add_trait = wounded_2
    }
    else_if = {
        limit = { has_trait = wounded_2 }
        remove_trait = wounded_2
        add_trait = wounded_3
    }
    else_if = {
        limit = { has_trait = wounded_3 }
        add_trait = maimed
    }
}
