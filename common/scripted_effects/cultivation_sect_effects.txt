join_mount_hua_effect = {
	if = {
		limit = { NOT = { has_character_flag = mount_hua_disciple } }
		clear_cultivation_sect_membership_effect = yes
		add_character_flag = joined_any_cultivation_sect
		add_character_flag = mount_hua_disciple
		add_character_flag = { flag = sect_switch_cooldown years = 15 }
	}
	add_character_modifier = { modifier = mount_hua_training months = 36 }
}
