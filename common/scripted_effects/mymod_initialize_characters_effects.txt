mymod_initialize_meridians_effect = {
    if = {
        limit = { NOT = { has_awakened_meridians = yes } }
        random_list = {
            100 = { add_trait = poor_meridians }
            70  = { add_trait = normal_meridians }
            40  = { add_trait = good_meridians }
            8   = { add_trait = excellent_meridians }
            1   = { add_trait = heavenly_meridians }
        }
    }
}

            NOT = { is_cultivator = yes }
        }

    }
}
