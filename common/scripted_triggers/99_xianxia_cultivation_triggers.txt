has_awakened_meridians = {
	OR = {
		has_trait = poor_meridians
		has_trait = normal_meridians
		has_trait = good_meridians
		has_trait = excellent_meridians
		has_trait = heavenly_meridians
	}
}

can_gain_qi_gathering = {
	is_adult = yes
	is_alive = yes
	NOT = { is_cultivator = yes }
	has_awakened_meridians = yes
}

is_reincarnator = {
	OR = {
		is_cultivator = yes
		has_trait = reincarnation
	}
}

is_reincarnation_eligible = {
	is_ai = no
	age < 11
	NOT = { has_trait = reincarnation }
	OR = {
		has_awakened_meridians = yes
		always = yes
	}
}
